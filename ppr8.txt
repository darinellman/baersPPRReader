


--03-06-20 - PPR's (all) (all pprs thru 03-05-20) (uph & case)

SELECT ITM.ITM_CD, ITM.VE_CD, ITM.VSN, ITM.ADV_PRC, ITM.RET_PRC, ITM.PRC1, ITM_FIFL.STORE_CD 
FROM INV.ITM ITM, INV.ITM_FIFL ITM_FIFL  
WHERE ITM.ITM_CD = ITM_FIFL.ITM_CD 
AND ((ITM.ADV_PRC BETWEEN 0.01 And 1000000.00) And ITM.ADV_PRC IS NOT NULL) 
AND (ITM_FIFL.STORE_CD BETWEEN '01' And '32') 
AND (ITM.ITM_CD BETWEEN '100000000' And '859985700')
AND (ITM.PRC3_CHNG_DT > DATE '2001-06-30')
ORDER BY ITM.VE_CD, ITM.VSN DESC;
