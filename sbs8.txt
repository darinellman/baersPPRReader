

SELECT SO.SO_STORE_CD, SO_LN.ITM_CD, SUM(SO_LN.QTY)
FROM INV.ITM ITM, SALES.SO SO, SALES.SO_LN SO_LN
WHERE SO_LN.ITM_CD = ITM.ITM_CD AND SO.DEL_DOC_NUM = SO_LN.DEL_DOC_NUM AND ((SO.STAT_CD<>'V')
AND (SO.SO_WR_DT Between DATE '2020-12-01' And DATE '2020-12-31')
AND (SO.ORD_TP_CD='SAL') AND (SO_LN.VOID_FLAG<>'Y')
AND (SO.SO_STORE_CD Between '23' AND '23')
AND (SO_LN.ITM_CD > '600000000')
AND (SO_LN.ITM_CD < '820999999'))
GROUP BY SO_LN.ITM_CD, SO.SO_STORE_CD
ORDER BY SO.SO_STORE_CD;
